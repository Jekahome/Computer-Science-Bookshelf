fn direct_to_xy(n: u32, mut d: u32) -> (u32, u32) {
    let mut x = 0;
    let mut y = 0;
    let mut s = 1; // –¢–µ–∫—É—â–∏–π —Ä–∞–∑–º–µ—Ä –ø–æ–¥–∫–≤–∞–¥—Ä–∞—Ç–∞ (—Ä–∞—Å—Ç–µ—Ç –æ—Ç 1 –¥–æ 2^(n-1))

    }
}

## Dancing Machine

> –ó–∞–¥–∞—á–∞:
> 
> –ù–∞ —ç—Ç–æ–º —É—Ä–æ–≤–Ω–µ –≤—ã –ø–æ–ª—É—á–∏—Ç–µ —Ç–æ–ª—å–∫–æ –æ–¥–∏–Ω –≤—Ö–æ–¥, –º—ã –Ω–∞–∑—ã–≤–∞–µ–º —ç—Ç–æ –Ω–∞—á–∞–ª—å–Ω—ã–º —Å–µ–º–µ–Ω–µ–º (`seed`). –°–µ–º—è –ø—Ä–æ—Ö–æ–¥–∏—Ç —á–µ—Ä–µ–∑ —Å–ª–µ–¥—É—é—â–∏–µ —à–∞–≥–∏ –¥–ª—è –ø–æ–ª—É—á–µ–Ω–∏—è –ø—Å–µ–≤–¥–æ-—Å–ª—É—á–∞–π–Ω–æ–≥–æ —á–∏—Å–ª–∞ (–ì–ü–°–ß).
> 
> –í –∞–ª–≥–æ—Ä–∏—Ç–º–µ –Ω–∏–∂–µ:
> * "shl 1" –æ–∑–Ω–∞—á–∞–µ—Ç —Å–¥–≤–∏–≥ –≤–ª–µ–≤–æ –æ–¥–∏–Ω —Ä–∞–∑
> * "shl 2" –æ–∑–Ω–∞—á–∞–µ—Ç —Å–¥–≤–∏–≥ –≤–ª–µ–≤–æ –¥–≤–∞–∂–¥—ã
> * "shr 1" –æ–∑–Ω–∞—á–∞–µ—Ç —Å–¥–≤–∏–≥ –≤–ø—Ä–∞–≤–æ –æ–¥–∏–Ω —Ä–∞–∑
> 
> ```
> 8-–º–∏ –±–∏—Ç–æ–≤—ã–π xorshift RNG:
> 
> temp1     = seed xor (seed shr 1)
> temp2     = temp1 xor (temp1 sh 1)
> next_seed = temp2 xor (temp2 shr 2)
> ```

> 
> (–û–±—Ä–∞—Ç–∏—Ç–µ –≤–Ω–∏–º–∞–Ω–∏–µ, —á—Ç–æ –Ω–∞—á–∞–ª—å–Ω–æ–µ —Å–µ–º—è –Ω–∏–∫–æ–≥–¥–∞ –Ω–µ –±—É–¥–µ—Ç 0)

    let mut seed: u8 = 29; 
    
    println!("–ù–∞—á–∞–ª—å–Ω–æ–µ —Å–µ–º—è: {:08b} ({})", seed, seed);
    println!("---------------------------------------");

    for i in 1..=10 {
        // 1: temp1 = seed xor (seed shr 1) 
        let shr_1 = seed >> 1;// DIV 2
        let temp1 = seed ^ shr_1;

        // 2: temp2 = temp1 xor (temp1 shl 1) 
        // –∑–∞–º–µ–Ω—è–µ–º shl 1 –Ω–∞ —Å–ª–æ–∂–µ–Ω–∏–µ (temp1 + temp1)
        // –∏–º–∏—Ç–∏—Ä–æ–≤–∞—Ç—å 8-–±–∏—Ç–Ω–æ–µ –ø–µ—Ä–µ–ø–æ–ª–Ω–µ–Ω–∏–µ –∫–∞–∫ –≤ LEG
        let shl_1 = temp1.wrapping_add(temp1); 
        let temp2 = temp1 ^ shl_1;

        // 3: next_seed = temp2 xor (temp2 shr 2) 
        let shr_2 = temp2 >> 2;// DIV 4
        seed = temp2 ^ shr_2;

        // 4: –ü–æ–ª—É—á–∞–µ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏–µ (next_seed % 4) 
        // –ò—Å–ø–æ–ª—å–∑—É–µ–º –ø–æ–±–∏—Ç–æ–≤–æ–µ –ò (AND 3), —á—Ç–æ–±—ã –æ—Å—Ç–∞–≤–∏—Ç—å —Ç–æ–ª—å–∫–æ 2 –Ω–∏–∂–Ω–∏—Ö –±–∏—Ç–∞
        let direction = seed & 3;

        println!("–®–∞–≥ {}: –ù–∞–ø—Ä–∞–≤–ª–µ–Ω–∏–µ = {}, –ù–æ–≤—ã–π seed = {:08b} ({})", 
                 i, direction, seed, seed);

        // –ü–æ —É—Å–ª–æ–≤–∏—é, –µ—Å–ª–∏ seed —Å—Ç–∞–Ω–µ—Ç 0, —Å—Ç–æ–ø
        if seed == 0 { break; }
    }

    println!("OUT: {} (–ï–¥–µ–º –∫ –∏—Å—Ç–æ—á–Ω–∏–∫—É)", source); // –∫ –∏—Å—Ç–æ—á–Ω–∏–∫—É

    println!("OUT: {} (–ï–¥–µ–º –∫ —Ü–µ–ª–∏)", dest); // –∫ —Ü–µ–ª–∏

fn hanoi(disk_nr: i32, source: u8, dest: u8, spare: u8) {
    if disk_nr == 0 {
        command(source, dest);
    } else {
        // 1. move(disk_nr - 1, source, spare, dest)
        // source=>source, spare=>dest, dest=>spare
        hanoi(disk_nr - 1, source, spare, dest);
        
        // 2. –ü–µ—Ä–µ–º–µ—â–∞–µ–º —Å–∞–º –¥–∏—Å–∫
        command(source, dest);
        
        // 3. move(disk_nr - 1, spare, dest, source)
        // spare=>source, dest=dest, source=>spare
        hanoi(disk_nr - 1, spare, dest, source);
    }
}

fn main() {
    // –í—Ö–æ–¥–Ω—ã–µ –¥–∞–Ω–Ω—ã–µ

    let source = 0;
    let dest = 2;
    let spare = 1;
    
    // –ó–∞–ø—É—Å–∫–∞–µ–º —Ä–µ–∫—É—Ä—Å–∏—é

}

## Water World

> –ó–∞–¥–∞—á–∞:
> 
> –ù–∞–º –Ω—É–∂–Ω–∞ –≤–∞—à–∞ –ø–æ–º–æ—â—å –≤ –ø–æ–∏—Å–∫–µ –ø–æ–¥—Ö–æ–¥—è—â–µ–≥–æ –º–µ—Å—Ç–∞ –¥–ª—è –≤–æ–¥–Ω–æ–π –≥–æ—Ä–∫–∏ ¬´–ü–∏—Ä–∞—Ç—Å–∫–∏–π —Å–ø—É—Å–∫¬ª. –í —á–∞—Å—Ç–Ω–æ—Å—Ç–∏, –Ω–∞–º –Ω—É–∂–Ω–æ –º–µ—Å—Ç–æ, —Å–ø–æ—Å–æ–±–Ω–æ–µ –≤–º–µ—Å—Ç–∏—Ç—å –±–æ–ª—å—à–æ–π –æ–±—ä–µ–º –≤–æ–¥—ã.
> 
> –õ–∞–Ω–¥—à–∞—Ñ—Ç —Å–æ—Å—Ç–æ–∏—Ç –∏–∑ 16 —Å—Ç–æ–ª–±—Ü–æ–≤. –ß—Ç–æ–±—ã –ø–æ–ª—É—á–∏—Ç—å –≤—ã—Å–æ—Ç—É –ª–∞–Ω–¥—à–∞—Ñ—Ç–∞ –≤ –∫–∞–∂–¥–æ–º —Å—Ç–æ–ª–±—Ü–µ —Å–ª–µ–≤–∞ –Ω–∞–ø—Ä–∞–≤–æ, –ø—Ä–æ—á—Ç–∏—Ç–µ –≤—Ö–æ–¥–Ω—ã–µ –¥–∞–Ω–Ω—ã–µ 16 —Ä–∞–∑.
> 
> –ó–∞—Ç–µ–º –≤ –∫–∞—á–µ—Å—Ç–≤–µ –æ—Ç–≤–µ—Ç–∞ –≤—ã–≤–µ–¥–∏—Ç–µ **–æ–±—â–∏–π –æ–±—ä–µ–º**, –∫–æ—Ç–æ—Ä—ã–π –º–æ–∂–µ—Ç –≤–º–µ—Å—Ç–∏—Ç—å –ª–∞–Ω–¥—à–∞—Ñ—Ç.


–°—É—Ç—å –≤ —Ç–æ–º, —á—Ç–æ –≤–æ–¥–∞ –∑–∞–¥–µ—Ä–∂–∏–≤–∞–µ—Ç—Å—è –≤ "—è–º–∞—Ö". –ß—Ç–æ–±—ã –ø–æ–Ω—è—Ç—å, —Å–∫–æ–ª—å–∫–æ –≤–æ–¥—ã –Ω–∞–¥ –∫–æ–Ω–∫—Ä–µ—Ç–Ω—ã–º —Å—Ç–æ–ª–±—Ü–æ–º, –Ω–∞–º –Ω—É–∂–Ω–æ –∑–Ω–∞—Ç—å —Å–∞–º—ã–π –≤—ã—Å–æ–∫–∏–π —Å—Ç–æ–ª–± —Å–ª–µ–≤–∞ –æ—Ç –Ω–µ–≥–æ –∏ —Å–∞–º—ã–π –≤—ã—Å–æ–∫–∏–π —Å—Ç–æ–ª–± —Å–ø—Ä–∞–≤–∞ –æ—Ç –Ω–µ–≥–æ.

–í–æ–¥–∞ –Ω–∞–¥ —Å—Ç–æ–ª–±—Ü–æ–º –Ω–µ –º–æ–∂–µ—Ç –±—ã—Ç—å –≤—ã—à–µ, —á–µ–º —Å–∞–º—ã–π –Ω–∏–∑–∫–∏–π –∏–∑ –¥–≤—É—Ö –µ–≥–æ "–∫—Ä–∞–π–Ω–∏—Ö" —Å–æ—Å–µ–¥–µ–π (–ª–µ–≤–æ–≥–æ –∏ –ø—Ä–∞–≤–æ–≥–æ –º–∞–∫—Å–∏–º—É–º–æ–≤). –î–≤–∏–≥–∞—è—Å—å —Å –¥–≤—É—Ö —Å—Ç–æ—Ä–æ–Ω, –º—ã –≤—Å–µ–≥–¥–∞ —É–≤–µ—Ä–µ–Ω—ã, —á—Ç–æ –µ—Å–ª–∏ –º—ã –æ–±—Ä–∞–±–∞—Ç—ã–≤–∞–µ–º –ª–µ–≤—É—é —Å—Ç–æ—Ä–æ–Ω—É, —Ç–æ –≥–¥–µ-—Ç–æ —Å–ø—Ä–∞–≤–∞ —Ç–æ—á–Ω–æ –µ—Å—Ç—å —Å—Ç–µ–Ω–∞ –Ω–µ –Ω–∏–∂–µ —Ç–µ–∫—É—â–µ–π max_L.

–ê–ª–≥–æ—Ä–∏—Ç–º "–î–≤–∞ —É–∫–∞–∑–∞—Ç–µ–ª—è": 

–ò–¥—Ç–∏ —Å –¥–≤—É—Ö —Å—Ç–æ—Ä–æ–Ω –æ–¥–Ω–æ–≤—Ä–µ–º–µ–Ω–Ω–æ.

1. –°—Ç–∞–≤–∏–º –æ–¥–∏–Ω —É–∫–∞–∑–∞—Ç–µ–ª—å (`L`) –Ω–∞ –Ω–∞—á–∞–ª–æ (0), –¥—Ä—É–≥–æ–π (`R`) –Ω–∞ –∫–æ–Ω–µ—Ü (15).
2. –•—Ä–∞–Ω–∏–º –ø–µ—Ä–µ–º–µ–Ω–Ω—ã–µ `max_L` (–º–∞–∫—Å–∏–º–∞–ª—å–Ω–∞—è –≤—ã—Å–æ—Ç–∞ —Å–ª–µ–≤–∞) –∏ `max_R` (–º–∞–∫—Å–∏–º–∞–ª—å–Ω–∞—è –≤—ã—Å–æ—Ç–∞ —Å–ø—Ä–∞–≤–∞).
3. **–õ–æ–≥–∏–∫–∞:**
    * –ï—Å–ª–∏ `height[L]` –º–µ–Ω—å—à–µ `height[R]`:
        * –ï—Å–ª–∏ `height[L]` –±–æ–ª—å—à–µ –∏–ª–∏ —Ä–∞–≤–µ–Ω `max_L`, –æ–±–Ω–æ–≤–ª—è–µ–º `max_L`.
        * –ò–Ω–∞—á–µ (–µ—Å–ª–∏ —Ç–µ–∫—É—â–∏–π —Å—Ç–æ–ª–± –Ω–∏–∂–µ –º–∞–∫—Å–∏–º—É–º–∞), –¥–æ–±–∞–≤–ª—è–µ–º –≤ –æ—Ç–≤–µ—Ç —Ä–∞–∑–Ω–∏—Ü—É: `max_L - height[L]`.
        * –°–¥–≤–∏–≥–∞–µ–º `L` –≤–ø—Ä–∞–≤–æ.
    * –ò–Ω–∞—á–µ (–µ—Å–ª–∏ `height[R]` –º–µ–Ω—å—à–µ –∏–ª–∏ —Ä–∞–≤–µ–Ω `height[L]`):
        * –ï—Å–ª–∏ `height[R]` –±–æ–ª—å—à–µ –∏–ª–∏ —Ä–∞–≤–µ–Ω `max_R`, –æ–±–Ω–æ–≤–ª—è–µ–º `max_R`.
        * –ò–Ω–∞—á–µ, –¥–æ–±–∞–≤–ª—è–µ–º –≤ –æ—Ç–≤–µ—Ç —Ä–∞–∑–Ω–∏—Ü—É: `max_R - height[R]`.
        * –°–¥–≤–∏–≥–∞–µ–º `R` –≤–ª–µ–≤–æ.

fn solve_water_world(ram: [u8; 16]) -> u32 {
    let mut i_left = 0;           
    let mut i_right = 15;         
    let mut max_l = 0;          
    let mut max_r = 0;          
    let mut total_water = 0;    

    while i_left <= i_right {
        // –õ–æ–≥–∏–∫–∞: –æ–±—Ä–∞–±–∞—Ç—ã–≤–∞–µ–º —Ç—É —Å—Ç–æ—Ä–æ–Ω—É, –≥–¥–µ –≤—ã—Å–æ—Ç–∞ –º–µ–Ω—å—à–µ,

                max_l = ram[i_left]; // –û–±–Ω–æ–≤–∏–ª–∏ –ª–µ–≤—ã–π –ø–∏–∫
            } else {
                total_water += (max_l - ram[i_left]) as u32; // –ù–∞—à–ª–∏ —è–º—É!

                max_r = ram[i_right]; // –û–±–Ω–æ–≤–∏–ª–∏ –ø—Ä–∞–≤—ã–π –ø–∏–∫

                total_water += (max_r - ram[i_right]) as u32; // –ù–∞—à–ª–∏ —è–º—É!
            }
            i_right -= 1;
        }
    }
    total_water
}
